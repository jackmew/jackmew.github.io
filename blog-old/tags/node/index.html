var sockets = [];

var s = net.Server(function(socket){
    sockets.push(socket);

    socket.on('data',function(d){
        for(var i = 0;i<sockets.length;i++){
            sockets[i].write(d);
        }
    });
    //remove dead socket
    socket.on('end',function(){

net = require('net');

var sockets = [];

var s = net.Server(function(socket){
    sockets.push(socket);

    socket.on('data',function(d){
        for(var i = 0;i<sockets.length;i++){
            if (sockets[i] == socket) continue;
            sockets[i].write(d);
        }
    });
    //remove dead socket
    socket.on('end',function(){
        var i = sockets.indexOf(socket);
        sockets.splice(i,1);
    });
});

s.listen(8000);

  1. 同一條規則(同一個function宣告or使用) 會只用先比對到的function

         app.get('/users/:id?', function(req, res, next){
         var id = req.params.id;
         if (id) {
             // do something
         } else {
             next();
         }
     });

     app.get('/users', function(req, res){
         // do something else
     });

function的部分可以:

one : 兩個參數 => function(req,res) =>不產生callback next()

two : 三個參數 => function(req,res,next) =>有callback next() 可以passing route control

three : 四個參數 => function(err,req,res,next) =>特別處理errorHandling使用

RequireJS

JavaScript本身就具備Require指令來載入模組,然而我們發現並不是所以模組都要在一開始就載入好的

RequireJs可以做到當下需要的模組再載入的這個想法也就是Asynchrnous Module
Defintion(AMD),並且確保所以相依(dependency)當載入才會執行

CommandJS和RequireJS有什麼不同

module & package

package比較像是組合一堆module之後的完成品

export : 公開模組

require : 取得模組

有點類似於java的class

//hello.js

function Hello() {
 var name;

 this.setName = function(thyName) {
   name = thyName;
 };

 this.sayHello = function() {
   console.log('Hello ' + name);
 };
};

module.exports = Hello;

有點類似於java new 實體

//gethello.js

var Hello = require('./hello');

hello = new Hello();
hello.setName('BYVoid');
hello.sayHello();

Javascript是函數式程式設計語言,並沒有class概念,function本身就是物件,在node又叫做module,module可以藉由export公開,require引用

package.json

CommonJS規定用來描述套件(package)的檔案

Node.js會先去找package.json裡面的main 決定要先執行哪個js 若是沒有package.json就會去找index.js or index.node
