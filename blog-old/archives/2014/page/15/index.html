net = require('net');

var sockets = [];

var s = net.Server(function(socket){
    sockets.push(socket);

    socket.on('data',function(d){
        for(var i = 0;i<sockets.length;i++)
            sockets[i].write(d);
    });
});

s.listen(8000);

new 一個array ,將大家的socket放在裡面

此時在另一個terminal就可以讀到共同的sockets data 然後data被write出來

在結束後 remove dead socket 

var sockets = [];

var s = net.Server(function(socket){
    sockets.push(socket);

    socket.on('data',function(d){
        for(var i = 0;i<sockets.length;i++){
            sockets[i].write(d);
        }
    });
    //remove dead socket
    socket.on('end',function(){
        var i = sockets.indexOf(socket);
        sockets.splice(i,1);
    });
});

s.listen(8000);

最後加上if (sockets[i] == socket) continue ;
來判斷若是自己的socket回應的資料 就skip掉 這樣就可以解決重複看到自己
打的文字的問題 

net = require('net');

var sockets = [];

var s = net.Server(function(socket){
    sockets.push(socket);

    socket.on('data',function(d){
        for(var i = 0;i<sockets.length;i++){
            if (sockets[i] == socket) continue;
            sockets[i].write(d);
        }
    });
    //remove dead socket
    socket.on('end',function(){
        var i = sockets.indexOf(socket);
        sockets.splice(i,1);
    });
});

s.listen(8000);

這樣就是個極小型的聊天室?!
