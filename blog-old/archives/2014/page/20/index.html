@Override
public String getUIDUserString(List<String> userUIDList, String condition,String functionName){

        String startStr ="";
        String lastStr ="";
        if(functionName.equals("queryOnePage")){

            startStr = "{ $query:{$and:[{uid:{$regex:'^"+condition+"'}},{deleteMark:'N'},{$or:[";
            lastStr =" ]} ]},$orderby:{uid:1} }";

        }else if(functionName.equals("queryCount")){
            startStr = "{ $and:[{uid:{$regex:'^"+condition+"'}},{deleteMark:'N'},{$or:[";
            lastStr =" ]} ]}";
        }
        StringBuilder sb = new StringBuilder();
        sb.append(startStr);
        for(int i=0;i<userUIDList.size();i++){
            sb.append("{uid:{$regex:'^"+userUIDList.get(i)+"'}}");
            if(i != (userUIDList.size()-1)){
                sb.append(",");
            }
        }
        sb.append(lastStr);
        String queryStr = sb.toString();

  $("#modifyUIDForm").validate({

        showErrors: function(errorMap, errorList) {
                if(errorList.length==0) return;
                var msg = "";
                for(var i=0; i<errorList.length; i++) {
                    var error = errorList[i];
                    var selector = "label[for='" + error.element.id + "']"
                    var label = $(selector).text();

                    msg = msg + label + ":" + error.message + "\n";
                }
                alert(msg);
            },
            onfocusout: false,
            onkeyup: false
  });

onfocusout:true => 當滑鼠移開 focus out的時候 => 就會一直validate 很煩 =>所以設成false
