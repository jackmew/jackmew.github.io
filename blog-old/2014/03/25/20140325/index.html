scope

dreamerslab scope

function photo(){
        var name = 'JJ';
        return{
                say_name : function(){
                        console.log(name);
                },
                rename : function(newName){
                        name = newName;
                }
        };
}

var pic = new photo;

pic.say_name();
pic.rename('JS');
pic.say_name();

這樣子寫法 模擬出JavaBean getter ,setter 只有在new出instance後 自己才可以去get set value

這樣子的方式又叫做封裝(encapsulation)

那javascript中用 閉包(closure) 達到跟封裝差不多的效果
